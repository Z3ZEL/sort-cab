```json
{
    "folders" :[
    {
        "name" : "A",
        "from" : "a",
        "to" : "a"
    },
    {
        "name" : "BCD",
        "from" : "b",
        "to" : "d"
    }
],
    "categories" : [
        "category1",
        "category2",
        "category3"
    ]
}

```
